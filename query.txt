select b.title as TITLE, b.air_date as AIR_DATE, a.actor as ACTOR, a.character_name as CHARACTER, c.role as ROLE, d.salary as SALARY from portrayal a, episode b, charac c, stars d where  a.episode = b.title and a.series_creator = d.creator and a.character_name = c.name and a.season = d.season and a.actor = d.actor and b.quality_rating >= 4 order by air_date, title, actor, character;
